Task 3: Compute the Social Health Fund (SHIF) using selection control structures (3 Hours)

SHIF Rates

Farmer500

Code Implementation

Private Function CalculateSHIF(occupation As String) As Integer
    Select Case occupation.ToLower()
        Case "student"
            Return 0
        Case "farmer"
            Return 500
        Case "teacher"
            Return 2000
        Case "doctor"
            Return 2500
        Case Else
            Return 3000
    End Select
End Function

Private Sub btnCalculate_Click(sender As Object, e As EventArgs) Handles btnCalculate.Click
    Dim shif As Integer = CalculateSHIF(txtOccupation.Text)
    MessageBox.Show("The SHIF for " & txtName.Text & " is Ksh " & shif, "SHIF Result")
End Sub

Task 5: Use functions to input details, determine SHIF and display output (3 Hours)

Description

Create two functions:

  1. CaptureDetails() Captures patient info.

  2. ValidateDetails() Validates input data.

Code Implementation

vbnet
Private Function CaptureDetails() As Dictionary(Of String, String)
    Dim patient As New Dictionary(Of String, String)
    patient("ID") = InputBox("Enter Patient ID:")
    patient("Name") = InputBox("Enter Full Name:")
    patient("Gender") = InputBox("Enter Gender:")
    patient("Age") = InputBox("Enter Age:")
    patient("Location") = InputBox("Enter Location:")
    patient("Phone") = InputBox("Enter Phone Number:")
    patient("Occupation") = InputBox("Enter Occupation:")
    Return patient
End Function

Private Function ValidateDetails(patient As Dictionary(Of String, String)) As Boolean
    For Each value In patient.Values
        If value = "" Then
            MessageBox.Show("All fields must be filled!", "Validation Error")
            Return False
        End If
    Next
    Return True
End Function

Private Sub btnFunctionSHIF_Click(sender As Object, e As EventArgs) Handles btnFunctionSHIF.Click
    Dim p As Dictionary(Of String, String) = CaptureDetails()
    If ValidateDetails(p) Then
        Dim shif As Integer = CalculateSHIF(p("Occupation"))
        MessageBox.Show("Patient " & p("Name") & vbCrLf & "SHIF: " & shif, "SHIF Computation")
    End If
End Sub

Imports System.Data.SqlClient

Public Class frmDatabase
    Dim con As New SqlConnection("Data Source=.;Initial Catalog=GenzDispensaryDB;Integrated Security=True")

    Private Sub btnSave_Click(sender As Object, e As EventArgs) Handles btnSave.Click
        Try
            con.Open()
            Dim shif As Integer = CalculateSHIF(txtOccupation.Text)
            Dim query As String = "INSERT INTO Patients VALUES(@ID, @Name, @Gender, @Age, @Location, @Phone, @Occupation, @SHIF)"
            Dim cmd As New SqlCommand(query, con)
            cmd.Parameters.AddWithValue("@ID", txtID.Text)
            cmd.Parameters.AddWithValue("@Name", txtName.Text)
            cmd.Parameters.AddWithValue("@Gender", txtGender.Text)
            cmd.Parameters.AddWithValue("@Age", txtAge.Text)
            cmd.Parameters.AddWithValue("@Location", txtLocation.Text)
            cmd.Parameters.AddWithValue("@Phone", txtPhone.Text)
            cmd.Parameters.AddWithValue("@Occupation", txtOccupation.Text)
            cmd.Parameters.AddWithValue("@SHIF", shif)

            cmd.ExecuteNonQuery()
            MessageBox.Show("Patient record saved successfully!", "Success")
        Catch ex As Exception
            MessageBox.Show("Error: " & ex.Message)
        Finally
            con.Close()
        End Try
    End Sub
End Class
